LSB

在大多数PNG问图像中，没个像素都由R、G、B三原色组成，每种颜色一般用8位数据表示，如果修改其最低位，人眼是不能区分出这种微笑的变化的，因此可以利用没个像素的R、G、B颜色的分量的最低有效位来隐藏信息，这样每个像素可以携带3位的信息

根据题目提示，可以猜到大概率考的是这个知识点，用stegsolve打开，

直接看图片的最低有效位，发现隐藏了一个.png文件，将它保存下来，后缀名改为.png，打开发现是二维码，扫描即得flag

zip伪加密

根据题目，就知道考的是zip伪加密，010editor打开，将50 4B 01 02 后的 09 00 改为 00 00，就可直接解压得到txt文件内flag

git的使用

要使用命令行将文件提交到 GitHub 仓库中，你需要执行以下步骤：

  1. 克隆仓库（如果没有）： 如果你还没有将仓库克隆到本地，你需要先使用 git clone 命令将仓库克隆到本地。例如：

    git clone https://github.com/your-username/your-repository.git

  2. 进入仓库目录： 使用 cd 命令进入克隆下来的仓库目录：

    cd your-repository

  3. 添加文件到暂存区： 使用 git add 命令将要提交的文件添加到 Git 的暂存区。你可以添加单个文件或者多个文件。例如：

    git add file2.txt

    或者你也可以使用 git add . 命令将当前目录下所有文件添加到暂存区。

  4. 提交文件到本地仓库： 使用 git commit 命令将添加到暂存区的文件提交到本地仓库，并添加提交消息说明。例如：

    git commit -m "Add new files"

  5. 推送文件到远程仓库： 最后，使用 git push 命令将本地仓库的提交推送到 GitHub 远程仓库。你需要指定远程仓库的名称和分支。例如：

    git push origin main

    其中 origin 是远程仓库的名称，main 是要推送的分支名称。

执行完以上步骤后，你的文件就会被提交到 GitHub 仓库中了。

高精度加减乘除

本质：模拟人工手算（竖式计算

注：

1.输出格式要统一，因为在实际应用中大部分会是加减乘除的混合运算,but为什么不用python（滑稽，c++输麻了

2.推荐用printf进行输出，因为当输出内容比较多时，printf比cout的速度快，可以节省大量运行时间（需要输入大量数据时同理，使用scanf的速度会比cin快

3.除了加法，其他三个都要考虑前导零的存在

4. size_t只可能大于零，无法小于或等于零

高精度减法

#include<iostream>
#include<vector>
#include<algorithm>
#include<string>
using namespace std;
bool judge(const string& a, const string& b) {
    return a.size() > b.size() || (a.size() == b.size() && a >= b);
}//字符串的比较方式是根据acill码依次比较字符大小
vector<int> sub(string a,string b){
    vector<int> ret;
    size_t i=0;
    while(i<a.length()){
        int x=10;
        x+=a[i]-'0';
        if(i<b.length())x-=b[i]-'0';
        if(x<10&&i+1<a.length())a[i+1]-=1;
        ret.push_back(x%10);
        i++;
    }
    return ret;
}
int main()
{
    string a,b;
    getline(cin,a);
    getline(cin,b);
    bool flag=judge(a,b);//判断a，b大小
    if(flag==0)swap(a,b);//保证a一定大于等于b
    reverse(a.begin(),a.end());
    reverse(b.begin(),b.end());
    auto RET=sub(a,b);
    if(flag==0)cout<<'-';
    while (RET.size() > 1 && RET.back() == 0)
    RET.pop_back();//删去前导0，防止出现类似34-34=00的情况

高精度乘法

方法一：这个理论上可适用于大整数×大整数，but由于是本人自己写的，经检验好像也没啥问题但是不排除有可能存在bug（雾

本质是直接一位一位地模拟乘法竖式计算

#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
using namespace std;
vector<vector<int>> mul(const string &a,const string &b){
    //a相当于竖式计算中下面的那一个数，b相当于上面的那一个数
    vector<vector<int>> Ret(a.length());
    //a有n位，那么就会进行n次乘法计算，产生n个后续需要相加的数；
    for(size_t i=0;i<a.length();i++){
        int t=0;
        for(size_t k=i;k>0;k--){
            Ret[i].push_back(0);
        }//补0，例如a的第三位和b相乘，就需要补两个0
        for(size_t j=0;j<b.length();j++){
            t+=(b[j]-'0')*(a[i]-'0');
            Ret[i].push_back(t%10);
            t/=10;
        }
        if(t)Ret[i].push_back(t);
    }
   return Ret;
}
vector<int> add(const vector<int> &RET,const vector<int> &Ret){
    int t=0;
    size_t i=0,j=0;
    vector<int> result;
    while(i<RET.size()||j<Ret.size()){
        if(i<RET.size())t+=RET[i];
        if(j<Ret.size())t+=Ret[j];
        result.push_back(t%10);
        t/=10;
        i++,j++;
    }
    if(t)result.push_back(t);
    return result;
}
int main()
{
    string a,b;
    cin>>a>>b;
    reverse(a.begin(),a.end());
    reverse(b.begin(),b.end());
    auto Ret=mul(a,b);
    vector<int> RET;
    for(size_t i=0;i<a.length();i++){
        RET=add(RET,Ret[i]);
    }//将得出的n个数相加
    while(RET.size()>1&&RET.back()==0){
        RET.pop_back();
    }//删除前导零
    for(int i=RET.size()-1;i>=0;i--){
        printf("%d",RET[i]);
    }
}

方法二：适用于大整数×小整数

这个计算方式不同于传统的竖式计算，它把小整数视作一位数，整体与大整数的每一位数相乘 

#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
vector<int> mul(const string &A,const int &b){
    int t=0;
    vector<int> ret;
    size_t i=0;
    while(i<A.size()){
        t+=(A[i]-'0')*b;
        ret.push_back(t%10);
        t/=10;
        i++;
    }
    //把b视作一位数，整个过程就按大整数×一位数的方式进行
    if(t)
        ret.push_back(t);
    return ret;
}
int main()
{
    string A;
    int b;
    cin>>A>>b;
    reverse(A.begin(),A.end());
    auto ret=mul(A,b);
    while(ret.size()>1&&ret.back()==0)
        ret.pop_back();//删除前导零
    for(int i=ret.size()-1;i>=0;i--){
        printf("%d",ret[i]);
    }
}

高精度除法

适用于大整数/小整数，除法与之前加减乘不一样，不需要倒序从个位开始计算

#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
vector<int> div(const string &A,const int &b,int &RET){
    vector<int> ret;
    int t=0;
    int i=A.size()-1;
    while(i>=0){
        t=t*10+(A[i]-'0');
        ret.push_back(t/b);
        t=t%b;
        i--;
    }
    RET=t;
    reverse(ret.begin(),ret.end());//ret是正序储存的，为了方便后续删除前导零，将其反转
    return ret;
}
int main()
{
    string A;
    int b;
    cin>>A>>b;
    if(b==0)return 0;
    reverse(A.begin(),A.end());
    int RET;//储存余数
    auto ret=div(A,b,RET);
    while(ret.size()>1&&ret.back()==0){
        ret.pop_back();
    }
    for(int i=ret.size()-1;i>=0;i--){
        printf("%d",ret[i]);
    }
    cout<<endl<<RET;
}
